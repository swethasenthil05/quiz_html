function updateButtons() { document.querySelector('button[onclick="prevQuestion()"]').disabled = currentQuestion === 0; document.querySelector('button[onclick="nextQuestion()"]').disabled = currentQuestion === questions.length - 1; } function nextQuestion() { if (currentQuestion < questions.length - 1) { currentQuestion++; showQuestion(currentQuestion); updateButtons(); } } function prevQuestion() { if (currentQuestion > 0) { currentQuestion--; showQuestion(currentQuestion); updateButtons(); } } // Initialize updateButtons();
